bds.event = new function(){

	//存放事件－回调关系
	//其格式为：
	//eventMap : {
	//
	//	'eventName1' : [
	//		{
	//			'prod':eventProduct1,
	//			'callback':callback1
	//		},
	//		{
	//			'prod':eventProduct2,
	//			'callback':callback2
	//		},
	//	],
	//
	//	'eventName2' : [...]
	//	...
	//}

	var eventMap = {};

	/*
	 * 绑定事件监听
	 * eventId 事件ID
	 * callback 事件回调函数
	 */
	function listen(eventId,callback){
		if (typeof callback == "function" || callback instanceof Function){
			var eventObj = getEvent(eventId);
			eventMap[eventObj['name']] = eventMap[eventObj['name']] || [] ;
			eventMap[eventObj['name']].push({
				'prod':eventObj.prod,
				'callback':callback
			});
		}
	}


	/*
	 * 取消绑定事件监听
	 * eventId 事件ID
	 * callback 事件回调函数
	 */
	function unlisten(eventId,callback){

		var eventObj = getEvent(eventId),
			eventFunctions = eventMap[eventObj['name']] || [],
			i = 0;

		while(i < eventFunctions.length){

			var eventFunction = eventFunctions[i];

			if (callback === eventFunction['callback'] && eventLike(eventObj['prod'],eventFunction['prod'])) {
				eventFunctions.splice(i,1);
			}else{
				i++;
			}

		}
	}


	/*
	 * 分发事件
	 * eventId 事件ID
	 * data 事件数据
	 */
	function dispatch(eventId,data){

		var eventObj = getEvent(eventId),
			eventFunctions = eventMap[eventObj['name']] || [],
			eventData = {
				'data':data,
				'eventId':eventId
			};

		for(var i = 0,len = eventFunctions.length; i < len ; i++){

			var eventFunction = eventFunctions[i];

			try {
				if (eventLike(eventFunction['prod'],eventObj['prod'])) {
					eventFunction['callback'].call(this,eventData);
				}
			}catch(e){}
		}

	}


	/*
	 * 判断当前事件能否适配给出的事件id
	 * comm 通配符
	 * cur 当前事件id
	 * @return 是否适配
	 */
	function eventLike(comm,cur){
		return new RegExp("^" + comm.replace(/\./g,"\\.").replace(/\*/g,".+") +"$").test(cur);
	}


	/*
	 * 根据事件ID返回事件对象
	 * eventId 事件ID
	 * @return 事件对象 {'prod':事件所属产品线,'name':事件名}
	 */
	function getEvent(eventId){
		var eventVars = eventId.match(/(.+)\.(.+)/);
		if (eventVars && eventVars[2]) {
			return {
				prod:eventVars[1],
				name:eventVars[2]
			};
		}else {
			return {};
		}
	}


	this.on = listen;
	this.off = unlisten;
	this.trigger = dispatch;

	this.events = eventMap;
};
